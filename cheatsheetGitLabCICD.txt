docker-compose.yml
generalna struktura:

services:          # list of containers I want
  my-container:    # name I give it (anything)
    image: ...     # what image to use
    restart: ...   # what to do if it crashes
    volumes:       # folders to mount
      - host/path:container/path
    environment:   # env variables (if needed)
      - VAR=value
    ports:         # port mapping (if needed)
      - "host:container"

moja struktura:

	services:
  		gitlab-runner:
  			image: gitlab/gitlab-runner:latest
    		restart:
			always
    		volumes: 
      			- ~/gitlab-runner/config:/etc/gitlab-runner
			- /var/run/docker.sock:/var/run/docker.sock
